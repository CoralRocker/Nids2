Example

   use raylib::prelude::*;
    let color_white = Color::from_hex("FFFFFF").unwrap();
    let color_black = Color::from_hex("000000").unwrap();
     
   assert_eq!(color_black, Color::BLACK);
   assert_eq!(color_white, Color::WHITE);
